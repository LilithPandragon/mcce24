    \setlength{\parindent}{0.4cm}%
    \setlength{\parskip}{0pt}%
}

\newcommand{\germanitemize}{%

% Deutsche Tabellenformatierung
\newcommand{\germantable}[3]{%
    \begin{table}[htbp]%

    \label{#2}%
    \begin{tabular}{#3}%
    \toprule%
}

\newcommand{\germanfigure}[2]{%

    \caption{#2}%
    \label{fig:#1}%
}
